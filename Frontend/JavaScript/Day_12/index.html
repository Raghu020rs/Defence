//CRUD Operations let users = [ { id: 1, name: "Alice", age: 25 }, { id: 2, name: "Bob", age: 30 }, { id: 3, name: "Charlie", age: 35 } ]; // Create function createUser(name, age) { const newUser = { id: users.length + 1, name: name, age: age }; users.push(newUser); return newUser; } // Read function readUsers() { return users; } // Update function updateUser(id, name, age) { const user = users.find(user => user.id === id); if (user) { user.name = name; user.age = age; return user; } return null; } // Delete function deleteUser(id) { const index = users.findIndex(user => user.id === id); if (index !== -1) { return users.splice(index, 1)[0]; } return null; } // Example Usage console.log("Initial Users:", readUsers()); console.log("Create User:", createUser("David", 28)); console.log("Updated Users:", readUsers()); console.log("Update User:", updateUser(2, "Bob Smith", 31));
